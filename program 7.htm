{

}

else if (i==5)

{

document.getElementById('errConfirm').innerHTML = "<span style='color: red'>Your passwords don't match!</span>";

else

document.getElementById(div).innerHTML = "OK!";

}

else

document.getElementById(div).innerHTML = "OK!";

}

}
